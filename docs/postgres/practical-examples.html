CREATE OR REPLACE FUNCTION create_order(
    p_user_id BIGINT,
    p_items JSONB,  -- æ ¼å¼: [{"product_id": 1, "quantity": 2}, ...]
    p_shipping_address JSONB
)
RETURNS TABLE(order_id BIGINT, order_no VARCHAR)
LANGUAGE plpgsql
AS $$
DECLARE
    v_order_id BIGINT;
    v_order_no VARCHAR(32);
    v_total NUMERIC(10,2) := 0;
    v_item JSONB;
    v_product_id BIGINT;
    v_quantity INT;
    v_price NUMERIC(10,2);
    v_stock INT;
BEGIN
    -- ç”Ÿæˆè®¢å•å·
    v_order_no := 'ORD' || to_char(NOW(), 'YYYYMMDD') || LPAD(nextval('orders_id_seq')::TEXT, 6, '0');

    -- åˆ›å»ºè®¢å•
    INSERT INTO orders (order_no, user_id, total_amount, shipping_address, status)
    VALUES (v_order_no, p_user_id, 0, p_shipping_address, 1)
    RETURNING id INTO v_order_id;

    -- å¤„ç†æ¯ä¸ªå•†å“
    FOR v_item IN SELECT * FROM jsonb_array_elements(p_items)
    LOOP
        v_product_id := (v_item->>'product_id')::BIGINT;
        v_quantity := (v_item->>'quantity')::INT;

        -- è·å–å•†å“ä¿¡æ¯å¹¶é”å®šè¡Œ
        SELECT price, stock INTO v_price, v_stock
        FROM products
        WHERE id = v_product_id AND status = 1
        FOR UPDATE;

        -- æ£€æŸ¥åº“å­˜
        IF v_stock < v_quantity THEN
            RAISE EXCEPTION 'å•†å“ % åº“å­˜ä¸è¶³ï¼Œå½“å‰åº“å­˜ï¼š%', v_product_id, v_stock;
        END IF;

        -- æ‰£å‡åº“å­˜
        UPDATE products
        SET stock = stock - v_quantity
        WHERE id = v_product_id;

        -- æ·»åŠ è®¢å•æ˜ç»†
        INSERT INTO order_items (
            order_id, product_id, product_name, price, quantity, subtotal,
            product_snapshot
        )
        SELECT
            v_order_id,
            p.id,
            p.name,
            v_price,
            v_quantity,
            v_price * v_quantity,
            jsonb_build_object('price', p.price, 'specifications', p.specifications)
        FROM products p
        WHERE p.id = v_product_id;

        -- ç´¯åŠ æ€»é‡‘é¢
        v_total := v_total + (v_price * v_quantity);
    END LOOP;

    -- æ›´æ–°è®¢å•æ€»é‡‘é¢
    UPDATE orders SET total_amount = v_total WHERE id = v_order_id;

    RETURN QUERY SELECT v_order_id, v_order_no;
END;
$$;
