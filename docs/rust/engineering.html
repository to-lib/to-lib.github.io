fn serialize_uppercase<S>(value: &str, serializer: S) -> Result<S::Ok, S::Error>

    S: Serializer,

where
    D: Deserializer<'de>,

    let s: String = Deserialize::deserialize(deserializer)?;
    s.parse().map_err(serde::de::Error::custom)
}
