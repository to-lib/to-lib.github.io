        children.remove(component);
    }
    
    @Override
    public String getName() {
        return name;
    }
    
    @Override
    public void display() {
        System.out.println("ğŸ“ " + name + "/");
        for (FileSystemComponent child : children) {
            child.display();

    public long getSize() {

        for (FileSystemComponent child : children) {
            totalSize += child.getSize();
        }
        return totalSize;
    }
}
